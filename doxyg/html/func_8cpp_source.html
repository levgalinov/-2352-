33bool Graph::bfs(int source, int sink, std::vector<int>& parent) {
34 std::vector<bool> visited(V, false); // Array for tracking visited vertices
35 std::queue<int> q;
36 q.push(source);
37 visited[source] = true;
38 parent[source] = -1;
39
40 while (!q.empty()) {
41 int u = q.front();
42 q.pop();
43
44 // We go along all the edges from the vertex u
45 for (int v = 0; v < V; v++) {
46 // If the vertex has not yet been visited and there is residual capacity
47 if (!visited[v] && capacity[u][v] - flow[u][v] > 0) {
48 q.push(v);
49 visited[v] = true;
50 parent[v] = u;
51 if (v == sink) return true;
52 }
53 }
54 }
55 return false;
56}

65int Graph::edmondsKarp(int source, int sink) {
66 int maxFlow = 0;
67 std::vector<int> parent(V);
68
69 // Repeat as long as there is a way
70 while (bfs(source, sink, parent)) {
71 int pathFlow = INT_MAX;
72
73 // Let's determine the minimum residual capacity along the found path
74 for (int v = sink; v != source; v = parent[v]) {
75 int u = parent[v];
76 pathFlow = std::min(pathFlow, capacity[u][v] - flow[u][v]);
77 }
78
79 // Update streams along the path
80 for (int v = sink; v != source; v = parent[v]) {
81 int u = parent[v];
82 flow[u][v] += pathFlow;
83 flow[v][u] -= pathFlow;
84 }
85
86 maxFlow += pathFlow;
87 }
88
89 return maxFlow;
90}
